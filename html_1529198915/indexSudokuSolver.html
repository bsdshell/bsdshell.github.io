            public void solver(int k) {
                if(k == numEmpty) {
                    printBoard();
                }
                else {
                    int c = array[k][0];
                    int r = array[k][1];
                    for(int i=1; i <= gridSize; i++) {
                        if(checkColRow(c, r, i) && checkSquare(c, r, i)) {
                            board[c][r] = i;
                            solver(k+1);
                            board[c][r] = 0;
                        }
                    }
                }
            }
