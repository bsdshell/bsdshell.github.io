{-| 
    polynomial addition
-}
(+:)::(Num a)=>[(a, a)] -> [(a, a)] -> [(a, a)]
(+:) [] _ = []
(+:) _ [] = []
(+:) (x:cs) (y:cy) = ((fst x) + (fst y), snd x) : (+:) cs cy
