Clone Single Linked List
1. Similar problem when inert node to Binary Search Tree
2. [curr]->next = new Node[n]
Node* cloneSLL(Node* root){
    Node* curr = root;
    Node* ncurr = NULL;
    Node* head = NULL;
    while(curr){
        if(head == NULL){
            head = ncurr = new Node(curr->data); 
            curr = curr->right;
        }
        else{
            ncurr->right = new Node(curr->data);    
            ncurr = ncurr->right;
            curr = curr->right; 
        }
    }
    return head; 
}
