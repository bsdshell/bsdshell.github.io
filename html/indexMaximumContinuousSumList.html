            public static int continuousSum(int[] arr){
                int max = 0;
                if(arr != null){
                    int len = arr.length;
                    int cur = 0;
                    for(int i=0; i< len; i++){
                        if(cur < 0)
                            cur = 0;

                        cur = cur + arr[i];
                        // compare: [curr value, curr accumulator, max] 
                        max = Math.max(max, arr[i]);
                        max = Math.max(max, cur);
                    }
                }
                return max;
            }
