For Loop in Swift
for i in 0..<4{
    print(i)
} // 0, 1, 2, 3

Reverse For Loop in Swift
for i in (0..<4).reversed(){
    print(i)
} // 3, 2, 1, 0

Stride in For Loop in Swift
for i in stride(from:3, through:0, by: -1){
    print(i) 
} // 3 2 1 0

Iteratethrough array in Swift
let list = [1, 2, 3]
for item in list{
    print(item)
}   // 1 2 3
