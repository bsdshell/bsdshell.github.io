/**
    * print all binary from zero to num-1 with length log2(num)
    * @param num maximum number 
    */
    public static void printAllBinary(int num) {
        String str = "";
        int binLen = (int)Math.ceil(Math.log(num)/Math.log(2));

        for(int j=0; j<num; j++){
            int n = j;
            for(int i=0; i<binLen; i++){
                if(n % 2 == 0)
                    str = "0" + str;
                else
                    str = "1" + str;

                n = n/2;
            } 
            Print.plb(str);
            str = "";
        } 
    }
