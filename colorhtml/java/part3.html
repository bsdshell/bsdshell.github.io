// move all zeros to the right side of array
    // [1, 0, 3, 0] => [1, 3, 0, 0]
    public static void partition(Integer[] arr){
        if(arr != null){
            int zeroIndex = 0;
            int len = arr.length;
            for(int i=0; i<len; i++){
                if(arr[i].intValue() > 0){
                    if(i != zeroIndex)
                        swap(arr, i, zeroIndex);
                    // make sure the zeroIndex is not out of bounds
                    if(zeroIndex < len-1)
                        zeroIndex++;
                }
            }
        }
    }
