public static Node prev = null;
    public static boolean isBST(Node r) {
        if(r != null) {
            if(!isBST(r.left))
                return false;
            if(prev != null && prev.data >= r.data)
                return false;
            prev = r;
            if(!isBST(r.right))
                return false;
        }
        return true;
    }
