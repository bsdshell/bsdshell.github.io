// [1 -> 26] = ['A' -> 'Z']
    // [0 -> 25]   ['A' -> 'Z'] 
    public static void excelNumSaveLength(int num) {
        int binLen = 1;
        if(num == 1) 
            binLen = 1;
        else 
            binLen = (int)Math.ceil(Math.log(num)/Math.log(26));

        for(int j=1; j<=num; j++){
            int n = j;
            String str = "";
            for(int i=1; i<=binLen; i++){
                String s = Aron.intToAlphabetUpper((n-1)%26);
                str = s + str;
                n = (n-1)/26;
            } 
            Print.plb(str);
        } 
    }
