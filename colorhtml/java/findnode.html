public static Node findNode(Node root, int n){
        if(root != null){
            if( root.data == n)
                return root;
            Node l = findNode(root.left, n);
            Node r = findNode(root.right, n);
            if(l != null)
                return l;
            else if(r != null)
                return r;
        }
        return null;
    }
