    public static void spiral2(int[][] arr, int k){
        if(arr != null && arr.length > 0){
            int h = arr.length;
            int w = arr[0].length;
            
            if(h - 2*k == 1){
                for(int i=k; i<w-k; i++){
                    Print.p(arr[k][i]);
                }
            }else if(w - 2*k == 1){
                for(int i=k; i<h-k; i++){
                    Print.p(arr[i][k]);
                }
            }else if(k < Math.min(h, w)/2){
                for(int i=k; i<w-1-k; i++){
                    Print.p(arr[k][i]);
                } 
                for(int i=k; i<h-1-k; i++){
                    Print.p(arr[i][w-1-k]);
                } 
                for(int i=k; i<w-1-k; i++){
                    Print.p(arr[h-1-k][w-1-i]);
                } 
                for(int i=k; i<h-1-k; i++){
                    Print.p(arr[h-1-i][k]);
                } 
                spiral2(arr, k+1);
            }
        }
    }
