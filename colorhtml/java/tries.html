class Node{
    Node[] array = new Node[2];
    public Node(){
    }
}

class Tries{
   Node root; 
   public Tries(){
       root = new Node();
   }
}

public static void add(Node r, int[] arr, int k){
    if( k < arr.length){
        if(r.array[arr[k]] == null)
            r.array[arr[k]] = new Node();
        add(r.array[arr[k]], arr, k+1); 
    }
}

public static boolean contains(Node r, int[] arr, int h){
    if(h < arr.length && r.array[arr[h]] != null){
        if(h + 1 == arr.length)
            return true;
        else 
            return contains(r.array[arr[h]], arr, h+1);
    }
    return false;
}
