// deserialize with stack
    public static NNode deserializeStack(Iterator<String> ite){
        Stack<NNode> stack = new Stack<NNode>(); 
        while(ite.hasNext()){
            String token = ite.next();
            if(!token.equals("#")){
                stack.push(new NNode(token));
            }else{
                if(stack.size() > 1){
                    NNode top = stack.pop();
                    NNode peek = stack.peek();
                    peek.list.add(top);
                }
            }
        }
        return stack.peek();
    }
