// foldr (-) 1 [2, 3, 4]  output = 2 
    public static int foldr(int n){
        if(n == 4){
            return n - 1;
        }
        else{
            return n - foldr(n + 1);
        }
    }
    // foldl (-) 1 [2, 3, 4] output = -8 
    public static int foldl(int n){
        if(n == 2){
            return 1 - n;
        }else{
            return foldl(n-1) - n;
        }
    }
