// merge two sorted array
    public static int[] merge(int[] arr1, int[] arr2)
    {
        int[] array = null;
        if( arr1 != null && arr2 != null)
        {
            int i=0;
            int j=0;
            int k=0;
            int len1 = arr1.length;
            int len2 = arr2.length;
            array = new int[len1+len2];
            while(k < len1 + len2) 
            {
                if( i >= len1) 
                {
                    array[k] = arr2[j];
                    j++;
                }
                else if( j >= len2)
                {
                    array[k] = arr1[i];
                    i++;
                }
                else
                {
                    if(arr1[i] < arr2[j])
                    {
                        array[k] = arr1[i];
                        i++;
                    }
                    else 
                    {
                        array[k] = arr2[j];
                        j++;
                    }
                }
                k++;
            }
        }
        return array;
    }
