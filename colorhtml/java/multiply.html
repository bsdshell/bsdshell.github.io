public static int[] multiply(int[] arr)
    {
        if(arr != null && arr.length > 1)
        {
            int len = arr.length;
            int[] A = new int[len];
            int[] B = new int[len];

            A[0] = 1;
            B[len-1] = 1;
            for(int i = 1; i < len; i++)
            {
                A[i] = arr[i-1]*A[i-1];
                B[len-1-i] = arr[len-1-i+1]*B[len-1-i+1];
            }
            
            for(int i = 0; i < len; i++)
                A[i] = A[i]*B[i];

            return A;
        }
        return null;
    }
