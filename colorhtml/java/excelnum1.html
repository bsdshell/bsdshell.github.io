// [1 -> 26] = ['A' -> 'Z']
    // [0 -> 25]   ['A' -> 'Z'] 
    public static void printExcel(int num){
        int base = 26;
        for(int i=1; i<=num; i++){
            int x = i;
            String str = "";
            if(x-1 == 0)
                str = 'A' + str;
            else{
                while( x > 0){
                    int r = (x-1) % base;
                    String s = Aron.intToAlphabetUpper(r);
                    str = s + str;
                    x = (x-1) / base;
                }
            }
            Print.plb(str);
        }
    }
