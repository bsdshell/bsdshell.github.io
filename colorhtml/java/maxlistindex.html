    static int maxList(Integer[] arr) {
        int tmp_start = 0;
        int end = 0;
        int start = 0;
        int max = 0;
        int sum = 0;
        for(int i=0; i<arr.length; i++) {
            if(sum == 0) {
                tmp_start = i;
            }

            if(sum + arr[i] > 0) {
                int m = Math.max(arr[i], sum + arr[i]);
                if(m > max) {
                    max = m;
                    start = tmp_start;
                    end = i;
                }
                sum += arr[i];
            } else {
                sum = 0;
            }
        }
        Print.p("start[" + start + "]");
        Print.p("end  [" + end + "]");
        return max;
    }
