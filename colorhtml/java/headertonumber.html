static int excelNum(String header){
        int sum = 0;
        if(header != null){
            int len = header.length(); 
            for(int i=0; i<len; i++){
                int n = header.charAt(len-1 - i) - 'A' + 1;
                sum += n*Math.pow(26, i);
            }
        }
        return sum;
    }
