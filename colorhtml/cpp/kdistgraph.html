void kdistGraph(int** arr, int k, int h, int w){
    if(k == 0){
        std::cout<<"["<<h<<"]"<<std::endl;
    }else if(k > 0){
        for(int i=0; i<w; i++){
            if(h != i && arr[h][i]){
                if(k - 1 == 0)
                    std::cout<<"["<<i<<"]"<<std::endl;
                else if(k > 0)
                    kdistGraph(arr, k-1, i, w);
            }
        }
    }
    
}
