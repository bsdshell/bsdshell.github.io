void cloneDLL(LNode* root, LNode*& head, LNode*& tail){
    LNode* curr = root;
    LNode* ncurr = NULL;
    while(curr){
        if(head == NULL){
            tail = head = ncurr = new LNode(curr->data);
            curr = curr->next;
        }else{
            tail = ncurr->next = new LNode(curr->data);
            ncurr->next->prev = ncurr;
            ncurr = ncurr->next;
            curr = curr->next;
        }
    }
}
