// str: aaa 
// pat: aa
int count_occurrences(string str, string pat){
    int len = pat.length();
    int k = 0;
    int start = 0;
    int i = 0;
    int count = 0;
    while(i < str.length()){
        if(str[i] == pat[k]){
            if(k == 0)
                start = i;
            k++;
            if(k == len){
                count++;
                k = 0;

                start++;
                i = start;
            }else{
                i++;
            }
        }else{
            // reset start
            start++;
            i = start;
        }

    }
    return count;
}
