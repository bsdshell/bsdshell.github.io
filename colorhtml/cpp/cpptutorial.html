    // allocate two dimension array in C
    int col = 4;
    int row = 3;
    int** ppt;
    
    // this is trick part
    ppt = (int**)malloc(sizeof(int*)*col);
    for(int j=0; j<col; j++)
    {
        ppt[j] = (int*)malloc(sizeof(int)*row);
    }

    int c = 0;
    for(int i=0; i<col; i++)
    {
        for(int j=0; j<row; j++)
        {
            ppt[i][j] = c++;
        }
    }

    for(int i=0; i<col; i++)
    {
        for(int j=0; j<row; j++)
        {
            cout<<"["<<i<<"]="<<ppt[i][j]<<" ";
        }
        cout<<endl;
    }

    for(int j=0; j<col; j++)
    {
        free(ppt[j]);
    }
