void permutate(int* array, int size, int k)
    {
        if(k == size)
        {
            for(int i=0; i<size; i++)
                std::cout<<"["<<array[i]<<"] ";
            std::cout<<std::endl;
        }
        else
        {
            for(int i=k; i<size; i++)
            {
                swap(array, i, k);

                for(int j=0; j<k; j++)
                    std::cout<<"[--]";
                std::cout<<"(i="<<i<<",k="<<k<<")"<<std::endl;
                permutate(array, size, i+1);
                swap(array, i, k);
            }
        }
    }
