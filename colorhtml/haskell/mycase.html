fun x = case x `mod` 2 of
            0 -> "even"
            _ -> "odd"

fun' x = answer 
            where answer 
                    | x < 10 = "less then 10"
                    | x == 10 = "equal 10"
                    | otherwise = case x `mod` 2 of 
                                        0 -> "greater than 10 even" 
                                        _ -> "greater than 10 and odd"
