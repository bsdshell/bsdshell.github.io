nameDo::IO String
nameDo = do putStr "What is your first name: "
            first <- getLine
            putStr "What is your last name: "
            last  <- getLine
            let full = first ++ " " ++ last
            putStrLn("full name " ++ full ++ " ")
            return full

main = nameDo >>= \name -> print name

-- alternative
main = do
    name <- nameDo
    print name
