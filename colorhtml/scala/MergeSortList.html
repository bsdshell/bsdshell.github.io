def merge(xs1: Array[Int], xs2:Array[Int]): Array[Int] = {
        val len1 = xs1.length 
        val len2 = xs2.length 
        
        var i = 0
        var j = 0
        var k = 0
        var xs3: Array[Int] = new Array[Int](len1+len2) 
        while(i < len1 || j < len2)
        {
            if(i >= len1){ 
                xs3(k) = xs2(j) 
                j += 1
            }else if(j >= len2){
                xs3(k) = xs1(i) 
                i += 1 
            }else{
                if(xs1(i) < xs2(j)){
                    xs3(k) = xs1(i)
                    i += 1
                }else{
                    xs3(k) = xs2(j)
                    j += 1
                }
            }
            k += 1
        }
        return xs3
    }
